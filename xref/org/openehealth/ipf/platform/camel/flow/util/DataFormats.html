16  package org.openehealth.ipf.platform.camel.flow.util;
17  
18  import java.io.ByteArrayInputStream;
19  import java.io.ByteArrayOutputStream;
20  
21  import org.apache.camel.Exchange;
22  import org.apache.camel.spi.DataFormat;
23  
24  /**
25   * Utility to format data using {@link DataFormat}s.
26   * 
27   * @author Martin Krasser
28   */
29  public class DataFormats {
30  
31      /**
32       * Marshals an object to a byte array.
33       * 
34       * @param object
35       *            object to be marshalled.
36       * @param exchange
37       *            message exchange.
38       * @param dataFormat
39       *            data format to use for marshalling.
40       * @return marshalled object.
41       * @throws DataFormatException
42       * 
43       * @see DataFormat#marshal(Exchange, Object, java.io.OutputStream)
44       */
45      public static byte[] marshal(Object object, Exchange exchange, DataFormat dataFormat) {
46          ByteArrayOutputStream stream = new ByteArrayOutputStream();
47          try {
48              dataFormat.marshal(exchange, object, stream);
49          } catch (Exception e) {
50              throw new DataFormatException("failed to marshal object", e);
51          }
52          return stream.toByteArray();
53      }
54      
55      /**
56       * Unmarshals an object from a byte array.
57       * 
58       * @param bytes byte array to be unmarshalled. 
59       * @param exchange
60       *            message exchange.
61       * @param dataFormat
62       *            data format to use for unmarshalling.
63       * @return unmarshalled object.
64       * @throws DataFormatException
65       * 
66       * @see DataFormat#unmarshal(Exchange, java.io.InputStream)
67       */
68      public static Object unmarshal(byte[] bytes, Exchange exchange, DataFormat dataFormat) {
69          ByteArrayInputStream stream = new ByteArrayInputStream(bytes);
70          try {
71              return dataFormat.unmarshal(exchange, stream);
72          } catch (Exception e) {
73              throw new DataFormatException("failed to unmarshal object", e);
74          }
75      }
76      
77  }
