16  package org.openehealth.ipf.platform.camel.flow;
17  
18  import org.apache.camel.Exchange;
19  import org.openehealth.ipf.commons.core.io.IOUtils;
20  
21  import java.io.IOException;
22  import java.io.Serializable;
23  import java.util.HashMap;
24  import java.util.Map;
25  
26  /**
27   * A serializable data container for {@link Exchange} content. The data
28   * contained in this packet are sufficient to re-create an {@link Exchange} and
29   * replay it with the flow manager.
30   * 
31   * @author Martin Krasser
32   */
33  public class PlatformPacket implements Serializable {
34  
35      private static final long serialVersionUID = -1727312493187004318L;
36  
37      private Map<String, Object> exchangeProperties;
38      
39      private Map<String, Object> messageProperties;
40      
41      private byte[] messageBody;
42      
43      private String replayStrategyId;
44      
45      /**
46       * Returns this packet's exchange properties.
47       * 
48       * @return this packet's exchange properties.
49       */
50      public Map<String, Object> getExchangeProperties() {
51          return exchangeProperties;
52      }
53  
54      /**
55       * Set this packet's exchange properties from a copy of {@link Exchange}
56       * properties.
57       * 
58       * @param exchangeProperties
59       *            copy of {@link Exchange} properties.
60       *            
61       * @see Exchange#getProperties()
62       */
63      public void setExchangeProperties(Map<String, Object> exchangeProperties) {
64          this.exchangeProperties = exchangeProperties;
65      }
66  
67      /**
68       * Returns this packet's message properties.
69       * 
70       * @return this packet's message properties.
71       */
72      public Map<String, Object> getMessageProperties() {
73          return messageProperties;
74      }
75  
76      /**
77       * Set this packet's message properties from a copy of
78       * {@link Exchange#getIn()} message properties.
79       * 
80       * @param messageProperties
81       *            copy of {@link Exchange#getIn()} message properties.
82       * 
83       * @see Exchange#getProperties()
84       */
85      public void setMessageProperties(Map<String, Object> messageProperties) {
86          this.messageProperties = messageProperties;
87      }
88  
89      /**
90       * Returns this packet's message body as byte array.
91       * 
92       * @return this packet's message body as byte array.
93       */
94      public byte[] getMessageBody() {
95          return messageBody;
96      }
97  
98      /**
99       * Returns this packet's message body as byte[].
100      * 
101      * @param messageBody
102      *            byte array representation of {@link Exchange#getIn()} message
103      *            body.
104      */
105     public void setMessageBody(byte[] messageBody) {
106         this.messageBody = messageBody;
107     }
108     
109     /**
110      * Returns the {@link ReplayStrategy} identifier.
111      * 
112      * @return the {@link ReplayStrategy} identifier.
113      */
114     public String getReplayStrategyId() {
115         return replayStrategyId;
116     }
117 
118     /**
119      * @param replayStrategyId
120      *          the replay strategy.
121      */
122     public void setReplayStrategyId(String replayStrategyId) {
123         this.replayStrategyId = replayStrategyId;
124     }
125 
126     /**
127      * Deserializes a {@link PlatformPacket} from a byte array.
128      * 
129      * @param packet
130      *            byte array representation of a {@link PlatformPacket}.
131      * @return a deserialized {@link PlatformPacket}.
132      */
133     public static PlatformPacket deserialize(byte[] packet) {
134         try {
135             return (PlatformPacket)IOUtils.deserialize(packet);
136         } catch (Exception e) {
137             throw new PlatformPacketException("cannot deserialize packet", e);
138         }
139     }
140     
141     /**
142      * Serializes this packet into a byte array.
143      * 
144      * @return the serialized packet.
145      */
146     public byte[] serialize() {
147         try {
148             return IOUtils.serialize(this);
149         } catch (Exception e) {
150             throw new PlatformPacketException("cannot serialize packet", e);
151         }
152     }
153  
154     /**
155      * Creates a copy of <code>map</code> omitting entries with values that
156      * are not {@link Serializable}.
157      * 
158      * @param map map to copy.
159      * @return serializable copy.
160      */
161     public static Map<String, Object> serializableCopy(Map<String, Object> map) {
162         HashMap<String, Object> result = new HashMap<>();
163         for (Map.Entry<String, Object> entry : map.entrySet()) {
164             if (Exchange.AGGREGATION_STRATEGY.equals(entry.getKey())){
165                 // The aggregation strategies are hold in a Map, which is serializable 
166                 // Do not serialize the aggregation streategies map. 
167                 // See org.apache.camel.Splitter#setAggregationStrategyOnExchange
168                 // See GroovyFlowTest for test case
169                 continue;
170             }
171             if (entry.getValue() instanceof Serializable) {
172                 try {
173                     // only way to proof if the object really is serializable.
174                     IOUtils.serialize(entry.getValue());
175                 } catch (IOException ioe){
176                     continue;
177                 }
178                 result.put(entry.getKey(), entry.getValue());
179             }
180         }
181         return result;
182     }
183     
184 }
