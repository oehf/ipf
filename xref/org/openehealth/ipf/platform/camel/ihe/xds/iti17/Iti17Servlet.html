35  /**
36   * Servlet implementation to dispatch incoming ITI-17 (Retrieve Document)
37   * requests.
38   * @author Jens Riemschneider
39   */
40  public class Iti17Servlet extends HttpServlet {
41      private static final long serialVersionUID = -401129606220792110L;
42      
43      @Override
44      protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
45      	CamelContext camelContext = getCamelContext();
46  
47          String requestURI = req.getRequestURI();        
48          String endpointName = toEndpointName(requestURI);
49          
50          Iti17Endpoint endpoint = camelContext.getEndpoint(endpointName, Iti17Endpoint.class);
51          if (endpoint == null) {
52              super.doGet(req, resp);
53              return;
54          }
55  
56          Iti17Consumer consumer = endpoint.getActiveConsumer();
57          if (consumer == null) {
58              super.doGet(req, resp);
59              return;
60          }
61          
62          String fullRequestURI = req.getRequestURL()
63              .append("?")
64              .append(req.getQueryString())
65              .toString();
66      
67          InputStream inputStream;
68          try {
69              inputStream = consumer.process(fullRequestURI);
70          }        
71          catch (Throwable e) {
72              resp.setStatus(500);
73              audit(false, req.getRemoteAddr(), fullRequestURI);
74              return;
75          }
76          
77          if (inputStream != null) {
78              try {
79                  resp.setStatus(200);
80                  IOUtils.copy(inputStream, resp.getOutputStream());
81              }
82              finally {
83                  IOUtils.closeQuietly(inputStream);
84                  audit(true, req.getRemoteAddr(), fullRequestURI);
85              }
86          }
87          else {
88              resp.setStatus(404);
89              audit(false, req.getRemoteAddr(), fullRequestURI);
90          }
91      }
92  
93      private String toEndpointName(String requestURI) {
94          String serviceName = requestURI.startsWith("/") ? requestURI.substring(1) : requestURI;
95          return "xds-iti17:" + serviceName;
96      }
97  
98      private CamelContext getCamelContext() {
99          ServletContext servletContext = getServletContext();
100         ApplicationContext appContext = WebApplicationContextUtils.getRequiredWebApplicationContext(servletContext);
101         Map<?, ?> camelContextBeans = appContext.getBeansOfType(CamelContext.class);
102         Validate.isTrue(camelContextBeans.size() == 1, "A single camelContext bean is required in the application context");
103         CamelContext camelContext = (CamelContext) camelContextBeans.values().iterator().next();
104         Validate.notNull(camelContext, "A single camelContext bean is required in the application context");
105         return camelContext;
106     }
107     
108     private void audit(boolean successful, String clientIpAddress, String documentUri) {
109         AuditorManager.getRepositoryAuditor().auditRetrieveDocumentEvent(
110                 successful ? RFC3881EventOutcomeCodes.SUCCESS : RFC3881EventOutcomeCodes.MAJOR_FAILURE,
111                 clientIpAddress,
112                 /*userName*/ null,
113                 documentUri, 
114                 /*documentUniqueId*/ null);
115     }

117 }
