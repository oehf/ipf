31  /**
32   * The Camel component for the ITI-15 transaction.
33   */
34  public class Iti15Component extends XdsComponent<XdsSubmitAuditDataset> {
35  
36  
37      public Iti15Component() {
38          super(ITI_15);
39      }
40  
41      @Override
42      protected Endpoint createEndpoint(String uri, String remaining, Map<String, Object> parameters) throws Exception {
43          return new XdsEndpoint<XdsSubmitAuditDataset>(uri, remaining, this,
44                  getCustomInterceptors(parameters),
45                  getFeatures(parameters),
46                  getSchemaLocations(parameters),
47                  getProperties(parameters),
48                  Iti15Service.class) {
49              @Override
50              public AbstractWsProducer<XdsSubmitAuditDataset, WsTransactionConfiguration, ?, ?> getProducer(AbstractWsEndpoint<XdsSubmitAuditDataset, WsTransactionConfiguration> endpoint,
51                                                                                                             JaxWsClientFactory<XdsSubmitAuditDataset> clientFactory) {
52                  return new Iti15Producer(endpoint, clientFactory);
53              }
54          };
55      }
56  
57  }
