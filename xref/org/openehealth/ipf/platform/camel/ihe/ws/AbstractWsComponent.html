16  package org.openehealth.ipf.platform.camel.ihe.ws;
17  
18  import org.apache.camel.impl.DefaultComponent;
19  import org.apache.cxf.feature.AbstractFeature;
20  import org.apache.cxf.interceptor.AbstractBasicInterceptorProvider;
21  import org.apache.cxf.interceptor.Interceptor;
22  import org.apache.cxf.interceptor.InterceptorProvider;
23  import org.apache.cxf.message.Message;
24  import org.openehealth.ipf.commons.ihe.ws.WsTransactionConfiguration;
25  import org.openehealth.ipf.commons.ihe.ws.cxf.audit.WsAuditDataset;
26  import org.openehealth.ipf.platform.camel.ihe.atna.AuditableComponent;
27  
28  import java.util.List;
29  import java.util.Map;
30  
31  /**
32   * Base component class for Web Service-based IHE components.
33  
34   * @param <AuditDatasetType> audit type
35   * @param <ConfigType> configuration type
36   *
37   * @author Dmytro Rud
38   */
39  abstract public class AbstractWsComponent<AuditDatasetType extends WsAuditDataset, ConfigType extends WsTransactionConfiguration>
40          extends DefaultComponent implements AuditableComponent<AuditDatasetType> {
41  
42      protected InterceptorProvider getCustomInterceptors(Map<String, Object> parameters) {
43          AbstractBasicInterceptorProvider provider = new AbstractBasicInterceptorProvider() {};
44          provider.setInInterceptors(castList(resolveAndRemoveReferenceListParameter(
45                  parameters, "inInterceptors", Interceptor.class)));
46          provider.setInFaultInterceptors(castList(resolveAndRemoveReferenceListParameter(
47                  parameters, "inFaultInterceptors", Interceptor.class)));
48          provider.setOutInterceptors(castList(resolveAndRemoveReferenceListParameter(
49                  parameters, "outInterceptors", Interceptor.class)));
50          provider.setOutFaultInterceptors(castList(resolveAndRemoveReferenceListParameter(
51                  parameters, "outFaultInterceptors", Interceptor.class)));
52          
53          return provider;
54      }
55  
56  	@SuppressWarnings({ "unchecked", "rawtypes" })
57  	private List<Interceptor<? extends Message>> castList(
58  			List<Interceptor> param) {
59  		return (List<Interceptor<? extends Message>>) (List<?>) param;
60  	}
61  
62      /**
63       * @return
64       *      static configuration parameters of the Web Service which
65       *      server endpoints of this transaction.
66       */
67      public abstract ConfigType getWsTransactionConfiguration();
68  
69  
70      protected List<AbstractFeature> getFeatures(Map<String, Object> parameters) {
71          return resolveAndRemoveReferenceListParameter(parameters, "features", AbstractFeature.class);
72      }
73  
74      protected List<String> getSchemaLocations(Map<String, Object> parameters) {
75          return resolveAndRemoveReferenceListParameter(parameters, "schemaLocations", String.class);
76      }
77  
78      protected Map<String, Object> getProperties(Map<String, Object> parameters) {
79          List<Map> mapList = resolveAndRemoveReferenceListParameter(parameters, "properties", Map.class);
80          return (mapList != null && mapList.size() == 1)? mapList.get(0) : null;
81      }
82  }
