38  /**
39   * Generic Web Service implementation for HL7 v3-based transactions.
40   * @author Dmytro Rud
41   */
42  @Slf4j
43  abstract public class AbstractHl7v3WebService extends AbstractWebService {
44  
45      private final Hl7v3WsTransactionConfiguration wsTransactionConfiguration;
46  
47      public AbstractHl7v3WebService(Hl7v3InteractionId hl7v3InteractionId) {
48          this((Hl7v3WsTransactionConfiguration)hl7v3InteractionId.getWsTransactionConfiguration());
49      }
50  
51      public AbstractHl7v3WebService(Hl7v3WsTransactionConfiguration wsTransactionConfiguration) {
52          Validate.notNull(wsTransactionConfiguration);
53          this.wsTransactionConfiguration = wsTransactionConfiguration;
54      }
55      
56      /**
57       * The proper message processing method.
58       * @param requestString
59       *      XML payload of the HL7 v3 request message.
60       * @return
61       *      XML payload of the HL7 v3 response message or an automatically generated NAK.
62       */
63      protected String doProcess(String requestString) {
64          Exchange result = process(requestString);
65          Exception exception = Exchanges.extractException(result);
66          if (exception != null) {
67              log.info("HL7 v3 service failed", exception);
68              return createNak(requestString, exception);
69          }
70          return Exchanges.resultMessage(result).getBody(String.class);
71      }
72  
73      /**
74       * Creates a transaction-specific NAK message.
75       */
76      protected String createNak(String requestString, Throwable throwable) {
77          return Hl7v3NakFactory.response(
78                  requestString,
79                  throwable,
80                  wsTransactionConfiguration.getNakRootElementName(),
81                  wsTransactionConfiguration.getControlActProcessCode(),
82                  false);
83      }
84  
85      /**
86       * Creates a transaction-specific NAK message.
87       */
88      protected String createNak(GPathResult request, Throwable throwable) {
89          return Hl7v3NakFactory.response(
90                  request, throwable,
91                  wsTransactionConfiguration.getNakRootElementName(),
92                  wsTransactionConfiguration.getControlActProcessCode(),
93                  false);
94      }
95  
96      public Hl7v3WsTransactionConfiguration getWsTransactionConfiguration() {
97          return wsTransactionConfiguration;
98      }
99  
100 
101     protected Hl7v3AuditDataset startAtnaAuditing(String requestString, Hl7v3AuditStrategy auditStrategy) {
102         Hl7v3AuditDataset auditDataset = null;
103         if (auditStrategy != null) {
104             try {
105                 auditDataset = auditStrategy.createAuditDataset();
106                 MessageContext messageContext = new WebServiceContextImpl().getMessageContext();
107                 HttpServletRequest servletRequest =
108                         (HttpServletRequest) messageContext.get(AbstractHTTPDestination.HTTP_REQUEST);
109                 if (servletRequest != null) {
110                     auditDataset.setClientIpAddress(servletRequest.getRemoteAddr());
111                 }
112                 auditDataset.setServiceEndpointUrl((String) messageContext.get(Message.REQUEST_URL));
113 
114                 AddressingProperties apropos = (AddressingProperties) messageContext.get(
115                                 JAXWSAConstants.ADDRESSING_PROPERTIES_INBOUND);
116                 if ((apropos != null) && (apropos.getReplyTo() != null) && (apropos.getReplyTo().getAddress() != null)) {
117                     auditDataset.setUserId(apropos.getReplyTo().getAddress().getValue());
118                 }
119 
120                 if (wsTransactionConfiguration.isAuditRequestPayload()) {
121                     auditDataset.setRequestPayload(requestString);
122                 }
123 
124                 auditStrategy.enrichAuditDatasetFromRequest(auditDataset, requestString, null);
125             } catch (Exception e) {
126                 log.error("Phase 1 of server-side ATNA auditing failed", e);
127             }
128         }
129         return auditDataset;
130     }
131 
132 
133     protected void finalizeAtnaAuditing(
134             Object response,
135             Hl7v3AuditStrategy auditStrategy,
136             Hl7v3AuditDataset auditDataset)
137     {
138         if (auditStrategy != null) {
139             try {
140                 auditStrategy.enrichAuditDatasetFromResponse(auditDataset, response);
141                 auditStrategy.doAudit(auditDataset);
142             } catch (Exception e) {
143                 log.error("Phase 2 of server-side ATNA auditing failed", e);
144             }
145         }
146     }
147 
148 }
