16  package org.openehealth.ipf.commons.ihe.core.atna.custom;
17  
18  import org.openhealthtools.ihe.atna.auditor.IHEAuditor;
19  import org.openhealthtools.ihe.atna.auditor.codes.rfc3881.RFC3881EventCodes;
20  import org.openhealthtools.ihe.atna.auditor.context.AuditorModuleContext;
21  import org.openhealthtools.ihe.atna.auditor.models.rfc3881.CodedValueType;
22  import org.openhealthtools.ihe.atna.auditor.utils.EventUtils;
23  
24  import java.util.Collection;
25  import java.util.List;
26  
27  import static org.openehealth.ipf.commons.ihe.core.atna.custom.CustomAuditorUtils.configureEvent;
28  
29  /**
30   * Implementation of HDM Auditors to send audit messages for
31   * <ul>
32   *     <li>ITI-59</li>
33   * </ul>
34   *
35   * @author Dmytro Rud
36   */
37  public class HpdAuditor extends IHEAuditor {
38  
39      public static HpdAuditor getAuditor() {
40          AuditorModuleContext ctx = AuditorModuleContext.getContext();
41          return (HpdAuditor) ctx.getAuditor(HpdAuditor.class);
42      }
43  
44      public void auditIti59(
45              boolean serverSide,
46              RFC3881EventCodes.RFC3881EventActionCodes eventActionCode,
47              RFC3881EventCodes.RFC3881EventOutcomeCodes eventOutcome,
48              String replyToUri,
49              String userName,
50              String directoryUri,
51              String clientIpAddress,
52              Collection<String> providerIds,
53              List<CodedValueType> purposesOfUse)
54      {
55          if (! isAuditorEnabled()) {
56              return;
57          }
58  
59          HpdEvent event = new HpdEvent(
60                  !serverSide,
61                  eventActionCode,
62                  eventOutcome,
63                  purposesOfUse);
64  
65          configureEvent(this, serverSide, event, replyToUri, userName, directoryUri, directoryUri, clientIpAddress);
66          if (! EventUtils.isEmptyOrNull(providerIds)) {
67              providerIds.forEach(event::addProviderParticipantObject);
68          }
69          audit(event);
70      }
71  
72  }
