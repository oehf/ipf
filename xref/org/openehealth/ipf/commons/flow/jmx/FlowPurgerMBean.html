16  package org.openehealth.ipf.commons.flow.jmx;
17  
18  import java.util.HashMap;
19  import java.util.Map;
20  
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  import org.openehealth.ipf.commons.flow.FlowManager;
24  import org.openehealth.ipf.commons.flow.config.ApplicationConfig;
25  import org.openehealth.ipf.commons.flow.purge.FlowPurgeJob;
26  import org.quartz.Scheduler;
27  import org.quartz.SchedulerException;
28  import org.quartz.impl.StdSchedulerFactory;
29  import org.springframework.beans.factory.DisposableBean;
30  import org.springframework.beans.factory.InitializingBean;
31  import org.springframework.beans.factory.annotation.Autowired;
32  import org.springframework.jmx.export.annotation.ManagedAttribute;
33  import org.springframework.jmx.export.annotation.ManagedOperation;
34  import org.springframework.jmx.export.annotation.ManagedResource;
35  
36  /**
37   * MBean for managing application-specific purge operations for out-dated flows.
38   * Out-dated flows are removed from the database either manually or periodically
39   * based on cron expressions.
40   * 
41   * @author Martin Krasser
42   */
43  @ManagedResource(
44          objectName="org.openehealth.ipf:type=service,name=FlowPurger",
45          description="Flow purge service") 
46  public class FlowPurgerMBean implements InitializingBean, DisposableBean {
47  
48      private static final Logger LOG = LoggerFactory.getLogger(FlowPurgerMBean.class);
49  
50      @Autowired
51      private FlowManager flowManager; 
52  
53      private String application; 
54  
55      private Scheduler scheduler;
56      
57      private final Map<String, FlowPurgeJob> flowPurgeJobs;
58  
59      public FlowPurgerMBean() {
60          flowPurgeJobs = new HashMap<>();
61      }
62  
63      @ManagedAttribute(description="Application name")
64      public String getApplication() {
65          return application;
66      }
67  
68      @ManagedAttribute(description="Application name")
69      public void setApplication(String application) {
70          this.application = application;
71      }
72  
73      @ManagedAttribute(description="Purge flows older than given duration "
74              + "(e.g. \"30d\" will purge flows older than 30 days)")
75      public String getPurgeFlowsOlderThan() {
76          return flowManager.getApplicationConfig(application).getPurgeFlowsOlderThan();
77      }
78  
79      @ManagedAttribute(description="Set to true to exclude ERROR flows from being purged. "
80          + "Set to false to purge CLEAN and ERROR flows")
81      public boolean isDoNotPurgeErrorFlows() {
82          return flowManager.getApplicationConfig(application).isDoNotPurgeErrorFlows();
83      }
84  
85      @ManagedAttribute(description="Cron expression for purge schedule")
86      public String getPurgeSchedule() {
87          return flowManager.getApplicationConfig(application).getFlowPurgeSchedule();
88      }
89  
90      @ManagedAttribute(description="Purge job status for current application")
91      public boolean isPurgeScheduled() {
92          return flowManager.getApplicationConfig(application).isFlowPurgeScheduled();
93      }
94  
95      @ManagedAttribute(description="Purge flows older than given duration "
96              + "(e.g. \"30d\" will purge flows older than 30 days)")
97      public void setPurgeFlowsOlderThan(String purgeFlowsOlderThan) {
98          ApplicationConfig applicationConfig = flowManager.getApplicationConfig(application);
99          String s = formatInput(purgeFlowsOlderThan);
100         if (s != null) {
101             applicationConfig.setPurgeFlowsOlderThan(s);
102             flowManager.mergeApplicationConfig(applicationConfig);
103         }
104     }
105 
106     @ManagedAttribute(description="Set to true to exclude ERROR flows from being purged. "
107         + "Set to false to purge CLEAN and ERROR flows")
108     public void setDoNotPurgeErrorFlows(boolean doNotPurgeErrorFlows) {
109         ApplicationConfig applicationConfig = flowManager.getApplicationConfig(application);
110         applicationConfig.setDoNotPurgeErrorFlows(doNotPurgeErrorFlows);
111         flowManager.mergeApplicationConfig(applicationConfig);
112     }
113 
114     @ManagedAttribute(description="Cron expression for purge schedule")
115     public void setPurgeSchedule(String purgeSchedule) {
116         ApplicationConfig applicationConfig = flowManager.getApplicationConfig(application);
117         applicationConfig.setFlowPurgeSchedule(formatInput(purgeSchedule));
118         flowManager.mergeApplicationConfig(applicationConfig);
119     }
120 
121     @ManagedOperation(description = "Executes a purge job once for current application")
122     public void execute() {
123         executeJob(flowManager.getApplicationConfig(application));
124     }
125 
126     @ManagedOperation(description = "Schedules or reschedules a purge job for current application and schedule")
127     public void schedule() {
128         scheduleJob(flowManager.getApplicationConfig(application));
129     }
130 
131     @ManagedOperation(description = "Unschedules a purge job for current application and schedule")
132     public void unschedule() {
133         unscheduleJob(flowManager.getApplicationConfig(application));
134     }
135 
136     @Override
137     public void afterPropertiesSet() throws Exception {
138         if (scheduler == null){
139     	    scheduler = StdSchedulerFactory.getDefaultScheduler();
140         }
141     	if (!scheduler.isStarted()){ 
142             scheduler.start();
143         }
144         LOG.info("Scheduler started: {}", scheduler.isStarted());
145         initJobs();
146     }
147 
148     @Override
149     public void destroy() throws Exception {
150         scheduler.shutdown();
151     }
152 
153     private void executeJob(ApplicationConfig config) {
154         FlowPurgeJob flowPurgeJob = getFlowPurgeJob(config);
155         flowPurgeJob.execute(config);
156     }
157 
158     private void scheduleJob(ApplicationConfig config) {
159         FlowPurgeJob flowPurgeJob = getFlowPurgeJob(config);
160         if (flowPurgeJob.isScheduled()) {
161             flowPurgeJob.unschedule();
162         }
163         flowPurgeJob.schedule(config);
164     }
165 
166     private void unscheduleJob(ApplicationConfig config) {
167         getFlowPurgeJob(config).unschedule();
168     }
169 
170     private void initJobs() {
171         LOG.info("Initialize flow purge jobs ... ");
172         for (ApplicationConfig config : flowManager.findApplicationConfigs()) {
173             if (config.isFlowPurgeScheduled()) {
174                 scheduleJob(config);
175             } else {
176                 LOG.info("Skip scheduling of job for application {}", config.getApplication());
177             }
178         }
179         LOG.info("Initialization done. ");
180     }
181 
182     private static String formatInput(String input) {
183         if (input == null) {
184             return null;
185         }
186         String trimmedInput = input.trim();
187         if (trimmedInput.isEmpty()) {
188             return null;
189         }
190         return trimmedInput;
191     }
192 
193     private FlowPurgeJob getFlowPurgeJob(ApplicationConfig config) {
194         FlowPurgeJob flowPurgeJob = flowPurgeJobs.get(config.getApplication());
195         if (flowPurgeJob == null) {
196             flowPurgeJob = new FlowPurgeJob(flowManager, scheduler, application);
197             flowPurgeJob.setFlowPurgeSchedule(config.getFlowPurgeSchedule());
198             flowPurgeJob.setPurgeFlowsOlderThan(config.getPurgeFlowsOlderThan());
199             flowPurgeJob.setDoNotPurgeErrorFlows(config.isDoNotPurgeErrorFlows());
200             flowPurgeJobs.put(application, flowPurgeJob);
201         }
202         return flowPurgeJob;
203     }
204 
205     public Scheduler getScheduler() {
206         return scheduler;
207     }
208 
209     public void setScheduler(Scheduler scheduler) {
210         this.scheduler = scheduler;
211     }
212 
213     @ManagedAttribute(description="Schedulers metadata")
214     public Map<String, String> getSchedulerMetaData() {
215         Map<String, String> map = new HashMap<>();
216         try {
217             map.put("Scheduler Name", scheduler.getMetaData().getSchedulerName());
218             map.put("Scheduler Class", scheduler.getMetaData().getSchedulerClass().getCanonicalName());
219             map.put("Thread Pool Class", scheduler.getMetaData().getThreadPoolClass().getCanonicalName());
220             map.put("Thread Pool Size", String.valueOf(scheduler.getMetaData().getThreadPoolSize()));
221             map.put("Number Of Jobs Executed", String.valueOf(scheduler.getMetaData().getNumberOfJobsExecuted()));
222             map.put("Running Since", MBeanUtils.formatDate(scheduler.getMetaData().getRunningSince()));
223             return map;
224         } catch (SchedulerException e) {
225             return null;
226         }
227     }
