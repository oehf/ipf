50  public class FlowRepositoryImpl extends HibernateDaoSupport implements FlowRepository {
51  
52      private static final Logger LOG = LoggerFactory.getLogger(FlowRepositoryImpl.class);
53  
54      @Autowired(required = false)
55      private SequenceRepository sequenceRepository;
56  
57      @Autowired(required = false)
58      private final FlowSearchCallback flowSearchCallback;
59      
60      public FlowRepositoryImpl() {
61          flowSearchCallback = new DefaultSearchCallback();
62      }
63      
64      @Override
65      public void initDao() throws Exception {
66          super.initDao();
67          if (sequenceRepository == null) {
68              sequenceRepository = new SequenceRepositoryMock();
69              LOG.warn("no sequence repository injected, using mock repository");
70          }
71          sequenceRepository.initSequence();
72      }
73  
74      @Override
75      public void persist(Flow flow) {
76          // Generate a sequence number for flow
77          flow.setIdentifier(sequenceRepository.nextNumber());
78          // Persist flow using the assigned number
79          getHibernateTemplate().persist(flow);
80      }
81  
82      @Override
83      public void merge(Flow flow) {
84          getHibernateTemplate().merge(flow);
85      }
86  
87      @Override
88      public void remove(Flow flow) {
89          getHibernateTemplate().delete(flow);
90      }
91  
92      void removeAll(Collection<Flow> flows) {
93          getHibernateTemplate().deleteAll(flows);
94      }
95  
96      @Override
97      public Flow find(Long id) {
98          Flow flow = getHibernateTemplate().get(Flow.class, id);
99          if (flow == null) {
100             throw new FlowException("no flow with id " + id);
101         }
102         return flow;
103     }
104 
105     @Override
106     public Flow lock(Long id) {
107         try {
108             return getHibernateTemplate().load(Flow.class, id, LockMode.UPGRADE);
109         } catch (HibernateObjectRetrievalFailureException e) {
110             throw new FlowException("no flow with id " + id);
111         }
112     }
113 
114     @Override
115     public int purgeFlows(FlowPurgeCriteria purgeCriteria) {
116         final List<Flow> purgeCandidates = findPurgeCandidates(purgeCriteria);
117         getHibernateTemplate().executeWithNativeSession(new HibernateCallback() {
118             @Override
119             public Object doInHibernate(Session session) throws HibernateException {
120                 for (Flow object : purgeCandidates){
121                     session.delete(object);
122                 }
123                 session.flush();
124                 return null;
125             }
126         });
127         return purgeCandidates.size();
128     }

206     private Object execute(FlowFinderCriteria flowFinderCriteria,
207             DetachedCriteria flowStatusCriteria, Session session,
208             boolean idProjection) {
209 
210         Criteria criteria = flowStatusCriteria.getExecutableCriteria(session);
211 
212         if (idProjection) {
213             criteria.setProjection(Projections.id());
214         }
215         
216         int maxResults = flowFinderCriteria.getMaxResults();
217         if (maxResults != FlowFinderCriteria.DEFAULT_MAX_RESULTS) {
218             criteria.setMaxResults(maxResults);
219         }
220 
221         if (flowFinderCriteria.hasMessageQuery()) {
222             FlowSearchCriteria flowSearchCriteria = new FlowSearchCriteria();
223             flowSearchCriteria.setHibernateCriteria(criteria);
224             flowSearchCriteria.setInboundMessageQuery(flowFinderCriteria.getInboundMessageQuery());
225             flowSearchCriteria.setOutboundMessageQuery(flowFinderCriteria.getOutboundMessageQuery());
226             return flowSearchCallback.findFlows(session, flowSearchCriteria);
227         } else {
228             return criteria.list();
229         }
230     }
231 
232     private Object execute(FlowPurgeCriteria flowPurgeCriteria,
233             DetachedCriteria flowStatusCriteria, Session session) {
234         Criteria criteria = flowStatusCriteria.getExecutableCriteria(session);
235 
236         int purgeCount = flowPurgeCriteria.getMaxPurgeCount();
237         if (purgeCount != FlowPurgeCriteria.DEFAULT_MAX_PURGE_COUNT) {
238             // Setting max results adds "fetch first 100 rows only" to the select query
239             // Works on Derby version 1.5.x and above with Hibernate version 3.5.x and above
240             criteria.setMaxResults(purgeCount);
241         }
242         return criteria.list();
243     }
244     
245     private static DetachedCriteria createFlowsCriteria(FlowFinderCriteria finderCriteria) {
246         DetachedCriteria criteria = DetachedCriteria.forClass(Flow.class)
247                 .setResultTransformer(Criteria.DISTINCT_ROOT_ENTITY)
248                 .setFetchMode("parts", FetchMode.JOIN) // eager
249                 .add(ge("creationTime", finderCriteria.getFrom()))
250                 .addOrder(Order.desc("identifier"));
251 
252         if (finderCriteria.getApplication() != null) {
253             // constrain query to a certain application name
254             criteria.add(eq("application", finderCriteria
255                     .getApplication()));
256         }
257         if (finderCriteria.getTo() != null) {
258             // there's an upper limit to creationTime property
259             criteria.add(Restrictions
260                     .le("creationTime", finderCriteria.getTo()));
261         }
262 
263         return criteria;
264     }
265 
266     private static DetachedCriteria createErrorFlowsCriteria(FlowFinderCriteria finderCriteria) {
267         return createFlowsCriteria(finderCriteria).createAlias("parts", "p").add(
268                 eq("p.status", FlowStatus.ERROR));
269     }
270 
271     private static DetachedCriteria createUnackFlowsCriteria(FlowFinderCriteria finderCriteria) {
272         return createFlowsCriteria(finderCriteria).add(
273                 isEmpty("parts"));
274     }
275 
276     private static DetachedCriteria createPurgeCriteria(FlowPurgeCriteria purgeCriteria) {
277         DetachedCriteria criteria = DetachedCriteria.forClass(Flow.class)
278             .setResultTransformer(Criteria.DISTINCT_ROOT_ENTITY)
279             // purge all flows with a creation time older than time limit
280             .add(lt("creationTime", purgeCriteria.getTimeLimit()))
281             // purge oldest flows first
282             .addOrder(Order.asc("creationTime"));
283 
284         if (purgeCriteria.getApplication() != null) {
285             criteria.add(eq("application", purgeCriteria.getApplication()));
286         }
287         
288         if (purgeCriteria.getPurgeMode() == PurgeMode.CLEAN) {
289             // omit flows with error status
290             criteria.add(
291                 or(
292                     eq("derivedStatus", FlowStatus.CLEAN),
293                     isNull("derivedStatus") // old flows (both CLEAN and ERROR)
294                 ) 
295             );
296             
297         }
298         return criteria;
299     }
