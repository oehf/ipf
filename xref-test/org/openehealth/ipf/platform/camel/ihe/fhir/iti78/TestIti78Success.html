58      @Test
59      public void testSendManualPdqm() {
60  
61          Bundle result = sendManually(familyParameters());
62  
63          // printAsXML(result);
64  
65          assertEquals(Bundle.BundleType.SEARCHSET, result.getType());
66          assertEquals(ResourceType.Bundle, result.getResourceType());
67          assertTrue(result.hasEntry());
68  
69  
70          PdqPatient p = (PdqPatient)result.getEntry().get(0).getResource();
71          assertEquals("Test", p.getName().get(0).getFamily().get(0).getValue());
72          assertEquals("http://localhost:8999/Patient/4711", p.getId());
73  
74  
75          // Check ATNA Audit
76  
77          MockedSender sender = getAuditSender();
78          assertEquals(1, sender.getMessages().size());
79          AuditMessage event = sender.getMessages().get(0).getAuditMessage();
80  
81          // Event
82          assertEquals(
83                  RFC3881EventCodes.RFC3881EventOutcomeCodes.SUCCESS.getCode().intValue(),
84                  event.getEventIdentification().getEventOutcomeIndicator());
85          assertEquals(
86                  RFC3881EventCodes.RFC3881EventActionCodes.EXECUTE.getCode(),
87                  event.getEventIdentification().getEventActionCode());
88          CodedValueType eventId = event.getEventIdentification().getEventID();
89          CodedValueType expectedEventId = new DICOMEventIdCodes.Query();
90          assertEquals(expectedEventId.getCode(), eventId.getCode());
91          assertEquals(expectedEventId.getCodeSystemName(), eventId.getCodeSystemName());
92          assertEquals(expectedEventId.getOriginalText(), eventId.getOriginalText());
93          CodedValueType eventTypeCode = event.getEventIdentification().getEventTypeCode().get(0);
94          CodedValueType expectedEventTypeCode = new CustomIHETransactionEventTypeCodes.PDQMQuery();
95          assertEquals(expectedEventTypeCode.getCode(), eventTypeCode.getCode());
96          assertEquals(expectedEventTypeCode.getCodeSystemName(), eventTypeCode.getCodeSystemName());
97          assertEquals(expectedEventTypeCode.getOriginalText(), eventTypeCode.getOriginalText());
98  
99          // ActiveParticipant Source
100         ActiveParticipantType source = event.getActiveParticipant().get(0);
101         assertTrue(source.isUserIsRequestor());
102         assertEquals("127.0.0.1", source.getNetworkAccessPointID());
103         assertEquals(RFC3881ActiveParticipantCodes.RFC3881NetworkAccessPointTypeCodes.IP_ADDRESS.getCode(), source.getNetworkAccessPointTypeCode());
104 
105         // ActiveParticipant Destination
106         ActiveParticipantType destination = event.getActiveParticipant().get(1);
107         assertFalse(destination.isUserIsRequestor());
108         assertEquals("http://localhost:" + DEMO_APP_PORT + "/Patient", destination.getUserID());
109         assertEquals("localhost", destination.getNetworkAccessPointID());
110 
111         // Audit Source
112         AuditSourceIdentificationType sourceIdentificationType = event.getAuditSourceIdentification().get(0);
113         assertEquals("IPF", sourceIdentificationType.getAuditSourceID());
114         assertEquals("IPF", sourceIdentificationType.getAuditEnterpriseSiteID());
115 
116         // Retrieved Patient
117 //        ParticipantObjectIdentificationType patient = event.getParticipantObjectIdentification().get(0);
118 //        assertEquals(RFC3881ParticipantObjectCodes.RFC3881ParticipantObjectTypeCodes.PERSON.getCode(), patient.getParticipantObjectTypeCode());
119 //        assertEquals(RFC3881ParticipantObjectCodes.RFC3881ParticipantObjectTypeRoleCodes.PATIENT.getCode(), patient.getParticipantObjectTypeCodeRole());
120 //        assertEquals("Patient/4711", patient.getParticipantObjectID());
121 
122         // Query Parameters
123         ParticipantObjectIdentificationType query = event.getParticipantObjectIdentification().get(0);
124         assertEquals(RFC3881ParticipantObjectCodes.RFC3881ParticipantObjectTypeCodes.SYSTEM.getCode(), query.getParticipantObjectTypeCode());
125         assertEquals(RFC3881ParticipantObjectCodes.RFC3881ParticipantObjectTypeRoleCodes.QUERY.getCode(), query.getParticipantObjectTypeCodeRole());
126         assertEquals("http://localhost:8999/Patient?family=Test", new String(query.getParticipantObjectQuery()));
127 
128         CodedValueType poitTypeCode = query.getParticipantObjectIDTypeCode();
129         assertEquals("ITI-78", poitTypeCode.getCode());
130         assertEquals("IHE Transactions", poitTypeCode.getCodeSystemName());
131         assertEquals("Mobile Patient Demographics Query", poitTypeCode.getOriginalText());
132         assertEquals("MobilePatientDemographicsQuery", query.getParticipantObjectID());
133     }

135     @Test
136     public void testGetResource() {
137         PdqPatient p = client.read()
138                 .resource(PdqPatient.class)
139                 .withId("4711")
140                 .execute();
141         assertEquals("Test", p.getName().get(0).getFamily().get(0).getValue());
142         assertEquals(String.format("http://localhost:%d/Patient/4711", DEMO_APP_PORT), p.getId());
143         MockedSender sender = getAuditSender();
144         assertEquals(1, sender.getMessages().size());
145         AuditMessage event = sender.getMessages().get(0).getAuditMessage();
146 
147         // Patient
148         ParticipantObjectIdentificationType poit = event.getParticipantObjectIdentification().get(0);
149         assertEquals(RFC3881ParticipantObjectCodes.RFC3881ParticipantObjectTypeCodes.PERSON.getCode(), poit.getParticipantObjectTypeCode());
150         assertEquals(RFC3881ParticipantObjectCodes.RFC3881ParticipantObjectTypeRoleCodes.PATIENT.getCode(), poit.getParticipantObjectTypeCodeRole());
151         assertEquals("Patient/4711", new String(poit.getParticipantObjectID()));
152     }
