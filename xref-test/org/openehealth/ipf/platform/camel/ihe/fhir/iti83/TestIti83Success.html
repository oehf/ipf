66      @Test
67      public void testSendManualPixm() {
68  
69          Parameters result = sendManuallyOnType(validQueryParameters());
70  
71          // printAsXML(result);
72  
73          Parameters.ParametersParameterComponent parameter = result.getParameter().iterator().next();
74          assertEquals(ResponseCase.getRESULT_VALUE(), ((Identifier)parameter.getValue()).getValue());
75  
76          // Check ATNA Audit
77          MockedSender sender = getAuditSender();
78          assertEquals(1, sender.getMessages().size());
79          AuditMessage event = sender.getMessages().get(0).getAuditMessage();
80          assertEquals(
81                  RFC3881EventCodes.RFC3881EventOutcomeCodes.SUCCESS.getCode().intValue(),
82                  event.getEventIdentification().getEventOutcomeIndicator());
83          assertEquals(
84                  RFC3881EventCodes.RFC3881EventActionCodes.EXECUTE.getCode(),
85                  event.getEventIdentification().getEventActionCode());
86          CodedValueType eventId = event.getEventIdentification().getEventID();
87          CodedValueType expectedEventId = new DICOMEventIdCodes.Query();
88          assertEquals(expectedEventId.getCode(), eventId.getCode());
89          assertEquals(expectedEventId.getCodeSystemName(), eventId.getCodeSystemName());
90          assertEquals(expectedEventId.getOriginalText(), eventId.getOriginalText());
91          CodedValueType eventTypeCode = event.getEventIdentification().getEventTypeCode().get(0);
92          CodedValueType expectedEventTypeCode = new CustomIHETransactionEventTypeCodes.PIXMQuery();
93          assertEquals(expectedEventTypeCode.getCode(), eventTypeCode.getCode());
94          assertEquals(expectedEventTypeCode.getCodeSystemName(), eventTypeCode.getCodeSystemName());
95          assertEquals(expectedEventTypeCode.getOriginalText(), eventTypeCode.getOriginalText());
96  
97          // ActiveParticipant Source
98          ActiveParticipantType source = event.getActiveParticipant().get(0);
99          assertTrue(source.isUserIsRequestor());
100         assertEquals("127.0.0.1", source.getNetworkAccessPointID());
101         assertEquals(RFC3881ActiveParticipantCodes.RFC3881NetworkAccessPointTypeCodes.IP_ADDRESS.getCode(), source.getNetworkAccessPointTypeCode());
102 
103         // ActiveParticipant Destination
104         ActiveParticipantType destination = event.getActiveParticipant().get(1);
105         assertFalse(destination.isUserIsRequestor());
106         assertEquals("http://localhost:" + DEMO_APP_PORT + "/Patient/$ihe-pix", destination.getUserID());
107         assertEquals("localhost", destination.getNetworkAccessPointID());
108 
109         // Patient (going in)
110         ParticipantObjectIdentificationType patientIn = event.getParticipantObjectIdentification().get(0);
111         assertEquals(RFC3881ParticipantObjectCodes.RFC3881ParticipantObjectTypeCodes.PERSON.getCode(), patientIn.getParticipantObjectTypeCode());
112         assertEquals(RFC3881ParticipantObjectCodes.RFC3881ParticipantObjectTypeRoleCodes.PATIENT.getCode(), patientIn.getParticipantObjectTypeCodeRole());
113         assertEquals("urn:oid:1.2.3.4|0815", new String(patientIn.getParticipantObjectID()));
114 
115         // Patient (going out)
116 //        ParticipantObjectIdentificationType patientOut = event.getParticipantObjectIdentification().get(1);
117 //        assertEquals(RFC3881ParticipantObjectCodes.RFC3881ParticipantObjectTypeCodes.PERSON.getCode(), patientOut.getParticipantObjectTypeCode());
118 //        assertEquals(RFC3881ParticipantObjectCodes.RFC3881ParticipantObjectTypeRoleCodes.PATIENT.getCode(), patientOut.getParticipantObjectTypeCodeRole());
119 //        assertEquals("http://org.openehealth/ipf/commons/ihe/fhir/2|4711", new String(patientOut.getParticipantObjectID()));
120 
121         ParticipantObjectIdentificationType query = event.getParticipantObjectIdentification().get(1);
122         assertEquals(RFC3881ParticipantObjectCodes.RFC3881ParticipantObjectTypeCodes.SYSTEM.getCode(), query.getParticipantObjectTypeCode());
123         assertEquals(RFC3881ParticipantObjectCodes.RFC3881ParticipantObjectTypeRoleCodes.QUERY.getCode(), query.getParticipantObjectTypeCodeRole());
124         assertEquals("http://localhost:8999/Patient/$ihe-pix?sourceIdentifier=urn%3Aoid%3A1.2.3.4%7C0815&targetSystem=urn%3Aoid%3A1.2.3.4.6", new String(query.getParticipantObjectQuery()));
125         CodedValueType poitTypeCode = query.getParticipantObjectIDTypeCode();
126         assertEquals("ITI-83", poitTypeCode.getCode());
127         assertEquals("IHE Transactions", poitTypeCode.getCodeSystemName());
128         assertEquals("Mobile Patient Identifier Cross-reference Query", poitTypeCode.getOriginalText());
129         assertEquals("PIXmQuery", query.getParticipantObjectID());
130     }
