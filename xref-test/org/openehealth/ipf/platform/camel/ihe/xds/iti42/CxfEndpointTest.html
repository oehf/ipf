71      @Before
72      public void setUp() throws IOException, URISyntaxException {
73          port = ServletServer.getFreePort();
74          server = new JettyServer();
75          server.setContextResource(getClass().getResource("/cxf-context.xml").toURI().toString());
76          server.setPort(port);
77          server.setContextPath("");
78          server.setServletPath("/*");
79          server.setServlet(new CXFServlet());
80  
81          server.start();
82      }
83  
84      @After
85      public void tearDown() {
86          server.stop();
87      }
88  
89      @Test
90      public void test() throws Exception {
91          runRequestAndExpectFailure();
92  
93          JaxWsServiceFactory<XdsAuditDataset> serviceFactory = new JaxWsRequestServiceFactory<>(
94                  ITI_42.getWsTransactionConfiguration(), "/iti-42", null, null, null);
95          ServerFactoryBean factory = serviceFactory.createServerFactory(MyIti42.class);
96          Server serviceServer = factory.create();
97  
98          runRequestAndExpectFailure();
99  
100         serviceServer.start();
101         Assert.assertEquals(Status.SUCCESS, runRequest().getStatus());
102 
103         serviceServer.stop();
104         runRequestAndExpectFailure();
105 
106         //serviceServer = factory.create();
107         serviceServer.start();
108         Assert.assertEquals(Status.SUCCESS, runRequest().getStatus());
109 
110         serviceServer.stop();
111     }
112 
113     private void runRequestAndExpectFailure() {
114         try {
115             runRequest();
116             Assert.fail("Expected Exception: " + RuntimeException.class);
117         }
118         catch (RuntimeException e) {
119             // ok
120         }
121     }
122 
123     private Response runRequest() {
124         JaxWsClientFactory clientFactory = new JaxWsRequestClientFactory<>(
125                 ITI_42.getWsTransactionConfiguration(),
126                 "http://localhost:" + port + "/iti-42",
127                 null, null, null, null, null);
128 
129         Iti42PortType client = (Iti42PortType) clientFactory.getClient();
130         RegisterDocumentSet request = SampleData.createRegisterDocumentSet();
131         EbXMLSubmitObjectsRequest ebXMLReq = reqTransformer.toEbXML(request);
132         SubmitObjectsRequest rawReq = (SubmitObjectsRequest) ebXMLReq.getInternal();
133         RegistryResponseType rawResp = client.documentRegistryRegisterDocumentSetB(rawReq);
134         EbXMLRegistryResponse30 ebXMLResp = new EbXMLRegistryResponse30(rawResp);
135         return respTransformer.fromEbXML(ebXMLResp);
136     }
137 
138     public static class MyIti42 implements Iti42PortType {
139         private final EbXMLFactory factory = new EbXMLFactory30();
140 
141         private final RegisterDocumentSetTransformer reqTransformer =
142             new RegisterDocumentSetTransformer(factory);
143 
144         private final ResponseTransformer respTransformer =
145             new ResponseTransformer(factory);
146 
147         private final SubmitObjectsRequestValidator reqValidator =
148             new SubmitObjectsRequestValidator();
149 
150         private final RegistryResponseValidator respValidator =
151             new RegistryResponseValidator();
152 
153         @Override
154         public RegistryResponseType documentRegistryRegisterDocumentSetB(SubmitObjectsRequest rawReq) {
155 
156             EbXMLSubmitObjectsRequest30 ebXMLReq = new EbXMLSubmitObjectsRequest30(rawReq);
157             reqValidator.validate(ebXMLReq, ITI_42);
158             RegisterDocumentSet request = reqTransformer.fromEbXML(ebXMLReq);
159 
160             Response response = new Response(Status.SUCCESS);
161             if (!request.getSubmissionSet().getEntryUuid().equals("submissionSet01")) {
162                 response.setStatus(Status.FAILURE);
163                 response.setErrors(Arrays.asList(new ErrorInfo(ErrorCode.REGISTRY_ERROR, "unexpected value", Severity.ERROR, null, null)));
164             }
165 
166             EbXMLRegistryResponse ebXMLResp = respTransformer.toEbXML(response);
167             respValidator.validate(ebXMLResp, ITI_42);
168             return (RegistryResponseType) ebXMLResp.getInternal();
169         }
170     }
