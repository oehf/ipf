43  /**
44   * Audit strategy which wraps a set of transaction-specific audit strategies and dispatches
45   * calls to one of them depending of which WSDL operation is currently being processed.
46   *
47   * @author Dmytro Rud
48   */
49  @Slf4j
50  public class DispatchAuditStrategy<T extends XdsAuditDataset> extends AuditStrategySupport<T> {
51  
52      private final Map<QName, XdsAuditStrategy<? extends XdsAuditDataset>> map;
53  
54      public DispatchAuditStrategy(Map<QName, XdsAuditStrategy<? extends XdsAuditDataset>> additionalMappings) {
55          super(true);
56          map = new HashMap<>();
57          map.put(new QName("urn:ihe:iti:xds-b:2007", "DocumentRegistry_RegistryStoredQuery"),
58                  new Iti18ServerAuditStrategy());
59          map.put(new QName("urn:ihe:iti:xds-b:2007", "RespondingGateway_CrossGatewayQuery"),
60                  new Iti38ServerAuditStrategy());
61          map.put(new QName("urn:ihe:iti:xds-b:2007", "RespondingGateway_CrossGatewayRetrieve"),
62                  new Iti39ServerAuditStrategy());
63          map.put(new QName("urn:ihe:iti:xds-b:2007", "DocumentRepository_ProvideAndRegisterDocumentSet-b"),
64                  new Iti41ServerAuditStrategy());
65          map.put(new QName("urn:ihe:iti:xds-b:2007", "DocumentRegistry_RegisterDocumentSet-b"),
66                  new Iti42ServerAuditStrategy());
67          map.put(new QName("urn:ihe:iti:xds-b:2007", "DocumentRepository_RetrieveDocumentSet"),
68                  new Iti43ServerAuditStrategy());
69          map.put(new QName("urn:ihe:iti:xds-b:2007", "DocumentRegistry_MultiPatientStoredQuery"),
70                  new Iti51AuditStrategy(true));
71          map.put(new QName("urn:ihe:iti:xds-b:2010", "DocumentRegistry_UpdateDocumentSet"),
72                  new Iti57ServerAuditStrategy());
73          map.put(new QName("urn:ihe:iti:xds-b:2007", "DocumentRegistry_RegisterOnDemandDocumentEntry"),
74                  new Iti61AuditStrategy(true));
75          map.put(new QName("urn:ihe:iti:xds-b:2010", "DocumentRegistry_DeleteDocumentSet"),
76                  new Iti62AuditStrategy(true));
77          map.put(new QName("urn:ihe:iti:xds-b:2007", "RespondingGateway_CrossGatewayFetch"),
78                  new Iti63AuditStrategy(true));
79          map.put(new QName("urn:ihe:rad:xdsi-b:2009", "DocumentRepository_RetrieveImagingDocumentSet"),
80                  new Rad69AuditStrategy(true));
81          map.put(new QName("urn:ihe:rad:xdsi-b:2009", "RespondingGateway_CrossGatewayRetrieveImagingDocumentSet"),
82                  new Rad75AuditStrategy(true));
83  
84          if (additionalMappings != null) {
85              map.putAll(additionalMappings);
86          }
87      }
88  
89      @Override
90      public T createAuditDataset() {
91          XdsAuditStrategy<? extends XdsAuditDataset> strategy = getAuditStrategy();
92          return (strategy != null) ? (T)strategy.createAuditDataset() : null;
93      }
94  
95      @Override
96      public T enrichAuditDatasetFromRequest(T auditDataset, Object request, Map<String, Object> parameters ) {
97          XdsAuditStrategy<T> strategy = (XdsAuditStrategy<T>)getAuditStrategy();
98          if (strategy != null) {
99              return strategy.enrichAuditDatasetFromRequest(auditDataset, request, parameters);
100         }
101         return null;
102     }
103 
104     @Override
105     public boolean enrichAuditDatasetFromResponse(T auditDataset, Object response) {
106         XdsAuditStrategy<T> strategy = (XdsAuditStrategy<T>)getAuditStrategy();
107         if (strategy != null) {
108             return strategy.enrichAuditDatasetFromResponse(auditDataset, response);
109         }
110         return false;
111     }
112 
113     @Override
114     public void doAudit(T auditDataset) {
115         XdsAuditStrategy<T> strategy = (XdsAuditStrategy<T>)getAuditStrategy();
116         if (strategy != null) {
117             strategy.doAudit(auditDataset);
118         }
119     }
120 
121     @Override
122     public RFC3881EventCodes.RFC3881EventOutcomeCodes getEventOutcomeCode(Object response) {
123         XdsAuditStrategy<? extends XdsAuditDataset> strategy = getAuditStrategy();
124         return (strategy != null) ? strategy.getEventOutcomeCode(response) : null;
125     }
126 
127     private XdsAuditStrategy<? extends XdsAuditDataset> getAuditStrategy() {
128         MessageContext messageContext = new WebServiceContextImpl().getMessageContext();
129         if ("GET".equals(messageContext.get(MessageContext.HTTP_REQUEST_METHOD))) {
130             return null;
131         }
132         if (! messageContext.containsKey(MessageContext.WSDL_OPERATION)) {
133             return null;
134         }
135 
136         QName operationName = (QName) messageContext.get(MessageContext.WSDL_OPERATION);
137         XdsAuditStrategy<? extends XdsAuditDataset> auditStrategy = map.get(operationName);
138         if (auditStrategy == null) {
139             log.debug("No strategy could be found for operation {}", operationName);
140         } else {
141             log.debug("Found strategy {} for operation {}", auditStrategy.getClass().getCanonicalName(), operationName);
142         }
143         return auditStrategy;
144     }
