79      private void testMarshal(String endpoint) throws Exception {
80          MessageAdapter<ADT_A01> message = inputMessage(resource);
81          mockOutput.expectedMessageCount(1);
82          producerTemplate.sendBody(endpoint, message);
83          mockOutput.assertIsSatisfied();
84          assertEquals(message.toString(), resultString());
85      }
86      
87      private void testUnmarshal(String endpoint) throws Exception {
88          InputStream stream = inputStream(resource);
89          mockOutput.expectedMessageCount(1);
90          producerTemplate.sendBody(endpoint, stream);
91          mockOutput.assertIsSatisfied();
92          assertEquals(inputMessage(resource).toString(), resultAdapter().toString());
93      }
94  
95      private String resultString() {
96          return resultMessage().getBody(String.class);
97      }
98  
99      @SuppressWarnings("unchecked")
100     private <T extends AbstractMessage> MessageAdapter<T> resultAdapter() {
101         return (MessageAdapter<T>)resultMessage().getBody();
102     }
103 
104     private Message resultMessage() {
105         return mockOutput.getExchanges().get(0).getIn();
106     }
107     
108     private static InputStream inputStream(String resource) throws IOException {
109         return Ghl7ExtensionTest.class.getResourceAsStream(resource);
110     }
111 
112     private static <T extends AbstractMessage> MessageAdapter<T> inputMessage(String resource) {
113         return MessageAdapters.load(resource);
114     }
